export type TestFunction = (value: any, m?: Model) => boolean;
export interface ValidationError {
  msg: string;
  value?: any;
}
 
export default class ValidationRule {
  prefix: string;
  msg: string;
  test: TestFunction;
  constructor(msg: string, test: TestFunction, prefix = "should") {
    this.test = test;
    this.msg = msg;
    this.prefix = prefix;
  }
  and(r: ValidationRule): ValidationRule {
    return new ValidationRule(`${this.msg} and ${r.msg}`, (v, m) => {
      return this.test(v, m) && r.test(v, m);
    });
  }
  or(r: ValidationRule): ValidationRule {
    return new ValidationRule(`${this.msg} or ${r.msg}`, (v, m) => {
      return this.test(v, m) || r.test(v, m);
    });
  }
  run(v: any, m?: Model): ValidationError | null {
    if (this.test(v, m)) return null;
    return {
      msg: `${this.prefix} ${this.msg}`,
      value: v,
    };
  }
}

export function length(min: number, max: number = Infinity): ValidationRule {
  return new ValidationRule(
    `be of length in range between ${min} and ${max}`,
    (v: any) => {
      let l = 0;
      if (!v) return false;
      switch (typeof v) {
        case "string":
          l = v.length;
          break;
        case "object":
          if (v instanceof Array) {
            l = v.length;
          } else {
            l = Object.keys(v).length;
          }
      }
      return l > min && l < max;
    }
  );
}
